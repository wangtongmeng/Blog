```js
    name: '',

    	access: ['super_admin'] // 当前路由可由哪些用户组浏览
  	}
    component: xxx,
    children: []
  }
]
```

```js
router.beforeEach((to, from, next) => {

      if (canTurnTo(to.name, user.access, routes)) next() // 有权限，可访问

})
```

## 动态挂载路由,页面和组件级别控制

### 在全局导航守卫中，根据权限动态生成路由

通过全局路由守卫进行权限控制

src/router/index.js，index.js 会在 main.js 入口文中执行

```js
import Vue from 'vue'
import Router from 'vue-router'
import { routes } from './router'
import store from '@/store'
import { setTitle, setToken, getToken } from '@/lib/util'
import clonedeep from 'clonedeep'

Vue.use(Router)

const router = new Router({
  routes
})

router.beforeEach((to, from, next) => {
  to.meta && setTitle(to.meta.title)
	// 获取 token
  const token = getToken()
  // 跳转路由根据 token 判断
  if (token) {
    // 若无权限规则，获取权限规则
    if (!store.state.router.hasGetRules) {
      // 获取用户信息，并拿到路由权限 rules
      store.dispatch('authorization').then(rules => {
        // 合并路由列表，过滤后的动态路由+基础路由
        store.dispatch('concatRoutes', rules).then(routers => {
          // 合并基础路由和根据权限生成的路由
          router.addRoutes(clonedeep(routers)) // 若使用深拷贝，会导致嵌套路由的一级组件不复用，所以建议不做深拷贝，把 vuex 的严格模式设备 false
          next({ ...to, replace: true })
        }).catch(() => {
          next({ name: 'login' })
        })
      })
    } else {
      // 若已有权限规则，继续
      next()
    }
  } else {
    // 若 token 不存在，跳转到登录页
    if (to.name === 'login') next()
    else next({ name: 'login' })
  }
})
```

### 请求用户信息获取权限数据

```js
import { login, authorization } from '@/api/user'
import { setToken } from '@/lib/util'

const state = {
  userName: 'Lison',
  rules: {} // 组件级别权限
}
const getters = {
  firstLetter: (state) => {
    return state.userName.substr(0, 1)
  }
}
const mutations = {
  SET_USER_NAME (state, params) {
    state.userName = params
  },
  SET_RULES (state, rules) {
    state.rules = rules
  }
}
const actions = {
  updateUserName ({ commit, state, rootState, dispatch }) {
    // rootState.appName
  },
  login ({ commit }, { userName, password }) {
    return new Promise((resolve, reject) => {
      login({ userName, password }).then(res => {
        if (res.code === 200 && res.data.token) {
          setToken(res.data.token)
          resolve()
        } else {
          reject(new Error('错误'))
        }
      }).catch(error => {
        reject(error)
      })
    })
  },
  // 根据 token 获取用户信息，其中包含权限规则
  authorization ({ commit }, token) {
    return new Promise((resolve, reject) => {
      authorization().then(res => {
        if (parseInt(res.code) === 401) {
          reject(new Error('token error'))
        } else {
          setToken(res.data.token)
          resolve(res.data.rules.page) // rules.page 路由级别权限
          commit('SET_RULES', res.data.rules.component) // rules.component 组件级别权限
        }
      }).catch(error => {
        reject(error)
      })
    })
  },
  logout () {
    setToken('')
  }
}

export default {
  getters,
  state,
  mutations,
  actions,
  modules: {
    //
  }
}
```

### 根据路由权限过滤动态路由逻辑

src/store/module/router.js

```js
import { routes, routerMap } from '@/router/router'

const state = {
  routers: routes, // 初始值就是基础路由列表，都可以访问
  hasGetRules: false
}

const mutations = {
  CONCAT_ROUTES (state, routerList) {
    state.routers = routerList.concat(routes) // 合并路由列表，注意，基础路由放在后面，因为404页面在最后
    state.hasGetRules = true
  }
}
// 根据路由权限对动态路由进行过滤
const getAccesRouterList = (routes, rules) => {
  return routes.filter(item => {
    if (rules[item.name]) {
      if (item.children) item.children = getAccesRouterList(item.children, rules)
      return true
    } else return false
  })
}

const actions = {
  // 合并路由列表
  concatRoutes ({ commit }, rules) {
    return new Promise((resolve, reject) => {
      try {
        let routerList = []
        // 若权限控制都为 true，直接使用
        if (Object.entries(rules).every(item => item[1])) {
          routerList = routerMap
        } else {
          // 若权限控制有 false 的，进行过滤
          routerList = getAccesRouterList(routerMap, rules)
        }
        commit('CONCAT_ROUTES', routerList) // 这一步好像不需要
        resolve(routerList) // 这里不要传 state.routers 后导致重复挂载
      } catch (err) {
        reject(err)
      }
    })
  }
}

export default {
  state,
  mutations,
  actions
}
```
